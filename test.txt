CREATE DATABASE edushare;
USE edushare;

CREATE TABLE kategori (
    id VARCHAR(20) PRIMARY KEY,
    nama_kategori VARCHAR(100) NOT NULL,
    slug VARCHAR(100) UNIQUE NOT NULL
); 

CREATE TABLE users (
    id VARCHAR(20) PRIMARY KEY,
    nama VARCHAR(100) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    no_telp VARCHAR(15),
    foto_profil LONGBLOB, -- Kolom baru untuk penyimpanan file gambar
    password VARCHAR(255) NOT NULL,
    role ENUM('user', 'admin') DEFAULT 'user',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
); 

CREATE TABLE course (
    id VARCHAR(20) PRIMARY KEY,
    nama_course VARCHAR(100) NOT NULL,
    deskripsi TEXT,
    created_by VARCHAR(20) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (created_by) REFERENCES users(id)
); 

CREATE TABLE materi (
    id VARCHAR(20) PRIMARY KEY,
    user_id VARCHAR(20) NOT NULL,
    course_id VARCHAR(20) NOT NULL,
    kategori_id VARCHAR(20) NOT NULL,
    judul VARCHAR(200) NOT NULL,
    deskripsi TEXT,
    file_materi VARCHAR(255) NOT NULL,
    tanggal_upload TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (course_id) REFERENCES course(id),
    FOREIGN KEY (kategori_id) REFERENCES kategori(id)
);

CREATE TABLE download (
    id VARCHAR(20) PRIMARY KEY,
    materi_id VARCHAR(20) NOT NULL,
    user_id VARCHAR(20) NOT NULL,
    tanggal_download TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (materi_id) REFERENCES materi(id),
    FOREIGN KEY (user_id) REFERENCES users(id)
);
